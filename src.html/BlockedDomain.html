  5 package net.ukrcom.cip_gov_ua_getter;
  6 
  7 import java.time.Instant;
  8 import java.time.LocalDateTime;
  9 import java.time.ZoneId;
 10 import java.time.format.DateTimeFormatter;
 11 
 12 public class BlockedDomain {
 13 
 14     protected String domainName;
 15     protected boolean isBlocked;
 16     protected LocalDateTime dateTime;
 17 
 18     /**
 19      * Конструктор класа.
 20      *
 21      * @param dn - ім'я домена
 22      * @param b - статус операції над доменом
 23      * @param dt - дата в форматі LocalDateTime
 24      */
 25     public BlockedDomain(String dn, boolean b, LocalDateTime dt) {
 26         this.domainName = dn;
 27         this.isBlocked = b;
 28         this.dateTime = dt;
 29     }
 30 
 31     /**
 32      * Конструктор класа.
 33      *
 34      * @param dn - ім'я домена
 35      * @param b - статус операції над доменом
 36      * @param s - дата в текстовому форматі. Береться з атрибута date.
 37      */
 38     public BlockedDomain(String dn, boolean b, String s) {
 39         this.domainName = dn;
 40         this.isBlocked = b;
 41         /*
 42         this.dateTime = LocalDateTime.parse(
 43                 s.replaceAll("Z$", "")
 44         );
 45          */
 46         this.dateTime = LocalDateTime.parse(s, DateTimeFormatter.ISO_DATE_TIME);
 47     }
 48 
 49     /**
 50      * Конструктор класа. Статус - блокування. Дата - початок епохи.
 51      *
 52      * @param dn - ім'я домена
 53      */
 54     public BlockedDomain(String dn) {
 55         this.domainName = dn;
 56         this.isBlocked = true;
 57         this.dateTime = Instant
 58                 .ofEpochMilli(0)
 59                 .atZone(
 60                         ZoneId.systemDefault()
 61                 )
 62                 .toLocalDateTime();
 63     }
 64 
 65     /**
 66      * Повертає ім'я домена.
 67      *
 68      * @return
 69      */
 70     public String getDomainName() {
 71         return domainName;
 72     }
 73 
 74     /**
 75      * Встановлює ім'я домена.
 76      *
 77      * @param s
 78      */
 79     public void setDomainName(String s) {
 80         this.domainName = s;
 81     }
 82 
 83     /**
 84      * Повертає статус.
 85      *
 86      * @return
 87      */
 88     public boolean getIsBlocked() {
 89         return this.isBlocked;
 90     }
 91 
 92     /**
 93      * Встановлює статус.
 94      *
 95      * @param b
 96      */
 97     public void setIsBlocked(boolean b) {
 98         this.isBlocked = b;
 99     }
100 
101     /**
102      * Повертає дату та час.
103      *
104      * @return
105      */
106     public LocalDateTime getDateTime() {
107         return this.dateTime;
108     }
109 
110     /**
111      * Встановлює дату та час з LocalDateTime.
112      *
113      * @param d
114      */
115     public void setDateTime(LocalDateTime d) {
116         this.dateTime = d;
117     }
118 
119     /**
120      * Встановлює дату та час, аналізуючи строку.
121      *
122      * @param s
123      */
124     public void setDateTime(String s) {
125         this.dateTime = LocalDateTime.parse(s);
126     }
127 
128     /**
129      * Повертає стан екземпляра класа в текстовому вигляді.
130      *
131      * @return
132      */
133     @Override
134     public String toString() {
135         return "[".concat(getDateTime().toString())
136                 .concat(getIsBlocked() ? " + " : " - ")
137                 .concat(getDomainName())
138                 .concat("]");
139     }
140 
141 }

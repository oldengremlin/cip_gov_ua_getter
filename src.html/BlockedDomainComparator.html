10 /**
11  * Компаратор для формування TreeSet з об'єктів jBlockedDomain. Порядок
12  * сортування задається по імені домена, як основному крітерію, далі по даті та
13  * часу видання розпорядження. Якщо ж дата та час збігаються то першим стає
14  * розпорядження, що блокує домен.
15  *
16  * @author olden
17  */
18 public class BlockedDomainComparator implements Comparator {
19 
20     @Override
21     public int compare(Object o1, Object o2) {
22         BlockedDomain d1 = (BlockedDomain) o1;
23         BlockedDomain d2 = (BlockedDomain) o2;
24         int compare = d1.getDomainName().compareTo(d2.getDomainName());
25         if (compare != 0) {
26             return compare;
27         }
28         compare = d1.getDateTime().compareTo(d2.getDateTime());
29         if (compare != 0) {
30             return compare;
31         }
32         return Boolean.compare(d1.getIsBlocked(), d2.getIsBlocked());
33     }

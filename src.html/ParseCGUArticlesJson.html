 5 package net.ukrcom.cip_gov_ua_getter;
 6 
 7 import org.json.JSONArray;
 8 import org.json.JSONException;
 9 import org.json.JSONObject;
10 
11 /**
12  * Клас, що розбирає JSON з переліком розпоряджень.
13  *
14  * @author olden
15  */
16 public class ParseCGUArticlesJson {
17 
18     protected final JSONObject json;
19     protected final JSONArray posts;
20 
21     /**
22      * Конструктор класа. Розбираємо сформований JSON з розпорядженнями.
23      *
24      * @param json_data
25      * @throws JSONException
26      */
27     public ParseCGUArticlesJson(String json_data) throws JSONException {
28         this.json = new JSONObject(json_data);
29         this.posts = json.getJSONArray("posts");
30     }
31 
32     /**
33      * Повертаємо масив з розібраними й відокремленими один від одного
34      * розпорядженнями, для подальшого аналізу.
35      *
36      * @return @throws JSONException
37      */
38     public JSONArray getPosts() throws JSONException {
39         return this.posts;
40     }
41 }
